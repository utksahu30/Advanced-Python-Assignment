def truckTour(petrolpumps):
    start = current= total =0
    f = len(petrolpumps)
    for i in range(f):
        petrol = petrolpumps[i][0]
        distance = petrolpumps[i][1]
        diff = petrol -distance
        current = current + diff
        total= total + diff
        if current<0:
            start = i+1
            current= 0
    if total >0:
        return start
    else: 
        return -1
